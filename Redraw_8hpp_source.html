332 inline RedrawList::RedrawList(const E& x, F draw, size_t ntotal, size_t nbuffer, size_t noffset)
333  : m_N(x.size()), m_ntot(ntotal), m_nbuf(nbuffer), m_noff(noffset), m_draw(draw)
334 {
335  QPOT_ASSERT(x.dimension() == 1);
336  QPOT_ASSERT(m_nbuf <= m_ntot);
337  QPOT_ASSERT(m_noff <= m_nbuf);
338  QPOT_ASSERT(2 * m_noff < m_ntot); // either redraw left or right, never both for the same "x"
339 
340  m_proximity = std::min(m_proximity, m_ntot);
341 
342  // draw yield distance (times two!), and convert to yield positions
343  m_val = m_draw({m_N, m_ntot});
344  m_pos = xt::cumsum(m_val, 1);
345 
346  // shift such that the mean of "x" is in the middle of the potential energy landscape
347  auto shift = xt::mean(m_pos, 1) - xt::mean(x);
348  for (size_t p = 0; p < m_N; ++p) {
349  xt::view(m_pos, p, xt::all()) -= shift(p);
350  }
351 
352  // register minimum and maximum positions for each particle
353  m_min = xt::view(m_pos, xt::all(), 0);
354  m_max = xt::view(m_pos, xt::all(), m_pos.shape(1) - 1);
355  QPOT_ASSERT(xt::all(x > m_min));
356  QPOT_ASSERT(xt::all(x < m_max));
357 
358  // allocate current index
359  m_idx = xt::empty<decltype(m_idx)::value_type>({m_N});
360  m_idx_t = xt::zeros<decltype(m_idx_t)::value_type>({m_N});
361  m_iredraw = xt::zeros<decltype(m_iredraw)::value_type>({m_N});
362  m_left = xt::empty<decltype(m_left)::value_type>({m_N});
363  m_right = xt::empty<decltype(m_right)::value_type>({m_N});
364 
365  // compute current index (updates "m_idx" only, "m_idx_t" is only changed at redraw)
366  for (size_t p = 0; p < m_N; ++p) {
367  size_t i = std::lower_bound(&m_pos(p,0), &m_pos(p,0) + m_pos.shape(1), x(p)) - &m_pos(p,0) - 1;
368  m_idx(p) = i;
369  m_left(p) = m_pos(p, i);
370  m_right(p) = m_pos(p, i + 1);
371  }
372 }

489 {
490  // store redraw
491  m_lock = true;
492  m_redraw = true;
493  xt::view(m_iredraw, xt::keep(index)) = +1;
494 
495  // allocate new yield distances (times two!)
496  xt::xtensor<double, 2> y = xt::empty<double>({index.size(), m_ntot});
497 
498  // apply buffer: insert previously drawn values
499  xt::view(y, xt::all(), xt::range(0, m_nbuf)) =
500  xt::view(m_val, xt::keep(index), xt::range(m_ntot - m_nbuf, m_ntot));
501 
502  // draw yield distances (times two!)
503  xt::view(y, xt::all(), xt::range(m_nbuf, m_ntot)) = m_draw({index.size(), m_ntot - m_nbuf});
504  QPOT_ASSERT(xt::all(y > 0));
505 
506  // store yield distances
507  xt::view(m_val, xt::keep(index), xt::all()) = y;
508 
509  // update yield positions
510  xt::view(y, xt::all(), 0) = xt::view(m_pos, xt::keep(index), m_ntot - m_nbuf);
511  xt::view(m_pos, xt::keep(index), xt::all()) = xt::cumsum(y, 1);

514  xt::noalias(m_min) = xt::view(m_pos, xt::all(), 0);
515  xt::noalias(m_max) = xt::view(m_pos, xt::all(), m_pos.shape(1) - 1);
516

555 template <class E>
556 inline void RedrawList::computeIndex(const E& x)
557 {
558  using index_type = decltype(m_idx)::value_type;
559  using position_type = typename E::value_type;
560 
561  if (!m_redraw) {
562  for (size_t p = 0; p < m_N; ++p)
563  {
564  index_type i = m_idx(p);
565  position_type xp = x(p);
566 
567  if (m_left(p) < xp && m_right(p) >= xp) {
568  continue;
569  }
570 
571  index_type l = i > m_proximity ? i - m_proximity : 0;
572  index_type r = std::min(i + m_proximity, m_ntot - 1);
573 
574  if (m_pos(p, l) < xp && m_pos(p, r) >= xp) {
575  i = std::lower_bound(&m_pos(p,l), &m_pos(p,l) + r - l, xp) - &m_pos(p,l) - 1 + l;
576  }
577  else {
578  i = std::lower_bound(&m_pos(p,0), &m_pos(p,0) + m_ntot, xp) - &m_pos(p,0) - 1;
579  }
580 
581  m_idx(p) = i;
582  m_left(p) = m_pos(p, i);
583  m_right(p) = m_pos(p, i + 1);
584  }
585  }
586  else {
587  for (size_t p = 0; p < m_N; ++p)
588  {
589  index_type i = m_idx(p);
590  position_type xp = x(p);
591  bool redraw = m_iredraw(p) != 0;
592 
593  if (m_left(p) < xp && m_right(p) >= xp && !redraw) {
594  continue;
595  }
596 
597  index_type l = i > m_proximity ? i - m_proximity : 0;
598  index_type r = std::min(i + m_proximity, m_ntot - 1);
599 
600  if (m_pos(p, l) < xp && m_pos(p, r) >= xp && !redraw) {
601  i = std::lower_bound(&m_pos(p,l), &m_pos(p,l) + r - l, xp) - &m_pos(p,l) - 1 + l;
602  }
603  else {
604  i = std::lower_bound(&m_pos(p,0), &m_pos(p,0) + m_ntot, xp) - &m_pos(p,0) - 1;
605  }
606 
607  m_idx(p) = i;
608  m_left(p) = m_pos(p, i);
609  m_right(p) = m_pos(p, i + 1);
610  }
611  }
612 
613  m_lock = false;
614 }
